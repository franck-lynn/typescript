视频播放

在 index.html 文件中, 加入:

```html
<script src="https://cdn.jsdelivr.net/npm/xgplayer@2.32.2/browser/index.min.js"></script>
```

西瓜视频播放器的cdn

在 video-player.vue 中, 就可以实现视频播放了

```vue
<script setup lang="ts">
import { onMounted, ref } from "vue"
import { useIntersectionObserver } from "@vueuse/core"

import poster from "./poster.png"

const disabled = ref(true)
const mounted = ref(false)

onMounted(() => {
  mounted.value = true
  // 1. 初始化大视频播放
  // 插件在 index.html 由 cdn 加载
  // @ts-ignore
  /* const play = */ new window.Player({
    id: "mse",
    url: "https://media.w3.org/2010/05/sintel/trailer.mp4",
    width: "100%",
    height: "100%",
    poster: poster,
  })
  // 2. 监听大视频播放器是否在视口区域
  /* const { stop } =  */ useIntersectionObserver(
    document.getElementById("video-container"), //
    (option) => {
      const [intersectionObserverEntry] = option
      const { isIntersecting } = intersectionObserverEntry!
      // 根据是否在视口区域决定小窗口视口播放
      disabled.value = isIntersecting
    }
  )
})
</script>
<template>
  <!-- 要传送到的小视频播放窗口 -->
  <div id="small"></div>

  <!-- 初始播放的大视频窗口 -->

  <div id="big-box">
    <!-- 要穿梭的DOM, to 到哪个属性 -->
    <Teleport to="#small" :disabled="disabled">
      <div id="mse"></div>
    </Teleport>
  </div>

  <!-- 监听出入视口的占位元素 -->
  <div id="video-container" class="observe-container"></div>

  <!-- 模拟滚动 -->
  <div style="height: 1500px"></div>
</template>

<style scoped>
#small {
  position: fixed;
  right: 20px;
  bottom: 20px;
  width: 200px;
  height: 140px;
  border: 1px solid red;
}
.observe-container {
  top: 600px;
  left: 420px;
  width: 600px;
  height: 420px;
}
#big-box {
  top: 0;
  left: 0;
  width: 600px;
  height: 420px;
  border: 1px solid red;
}
</style>

```
